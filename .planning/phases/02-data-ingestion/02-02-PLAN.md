    ```python
    class HealthImporterBase:
        """Base class for health data importers."""

        PLATFORM = "unknown"  # Subclass overrides: "google-fit", "huawei-health", "xiaomi"

        def __init__(self, data_dir=None, person_id=None):
            self.person_id = person_id
            self._data_dir = data_dir

        def import_export(self, export_path: str, start_date=None, end_date=None) -> dict:
            """Main entry: extract ZIP -> discover files -> parse -> dedup -> store.
            Returns {"imported": N, "skipped_dupes": N, "errors": [], "by_type": {}}"""

        def _extract_zip(self, zip_path: str) -> Path:
            """Extract ZIP to temp dir, return extraction root."""

        def _discover_files(self, extracted_dir: Path) -> list[Path]:
            """Subclass: return list of data files to process."""
            raise NotImplementedError

        def _parse_records(self, data_file: Path) -> list[dict]:
            """Subclass: parse file into list of {type, timestamp, data, _meta}."""
            raise NotImplementedError

        def _fuzzy_dedup(self, new_record: dict, existing_records: list[dict]) -> bool:
            """Return True if new_record is a fuzzy duplicate of any existing.
            Uses 60s timestamp window and 5% value tolerance per D-10.
            Pre-filters by record_type and same-day range per Pitfall 3."""

        def _store_records(self, records: list[dict]) -> dict:
            """Write records via HealthDataStore with dedup.
            Groups records by type, maps type to skill_name via _type_to_skill mapping.
            Returns counts dict."""

        def _normalize_timestamp(self, ts_str: str, tz_offset: str | None = None) -> str:
            """Normalize timestamp to ISO 8601 with timezone per Pitfall 4."""
    ```

    Key implementation details:
    - Per D-10: fuzzy dedup uses 60s timestamp window + 5% value tolerance, first-imported wins

    - Per Pitfall 4: normalize all timestamps to ISO 8601 with timezone offset, convert to UTC for comparison
    - Per D-08: supported metric types mapped to skill names: heart_rate -> heart-rate-tracker, steps -> step-tracker, sleep_session -> sleep-tracker, weight -> weight-tracker, bp -> blood-pressure-tracker, activity -> activity-tracker
    - _meta must include: source="import", device=self.PLATFORM, import_timestamp=now
    - Use relative imports within _shared (from .health_data_store import HealthDataStore)
    - Use `from __future__ import annotations` at top
